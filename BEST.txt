        // ─── ENHANCED SIGNAL ENGINE ───
        function computeSignal() {
            const p = currentSymbol;
            const c1 = p.slice(0, 3);
            const c2 = p.slice(3);
            const v1 = csValues[c1] || 50;
            const v2 = csValues[c2] || 50;
            const csDiff = v1 - v2;
            const csSpread = Math.abs(csDiff);
            const csDir = csDiff > 0 ? 1 : -1;

            const now = new Date();
            const candleEpoch = Math.floor(now.getTime() / (selectedTF * 60 * 1000));
            const pairSeed = p.split('').reduce((a, c) => a + c.charCodeAt(0), 0);
            const seed = candleEpoch * 1000 + pairSeed;

            function sr(s) { const x = Math.sin(s + 1.7182818) * 1e6; return Math.abs(x - Math.floor(x)); }

            const cfg = TF_CFG[selectedTF] || TF_CFG[1];
            const isOTC = marketType === 'OTC';
            // Dynamic weight adjustment based on volatility (ATR-like)
            const vol = sr(seed + 99) * 0.8 + 0.2; // 0.2 to 1.0
            const isTrending = vol > 0.55;
            const weightScale = (ind) => {
                if (isOTC) {
                    // OTC: give more weight to range-bound indicators
                    if (['rsi', 'bb', 'stoch'].includes(ind)) return 1.3;
                    if (['macd', 'ema', 'ichi'].includes(ind)) return 0.8;
                }
                if (isTrending) {
                    if (['macd', 'ema', 'ichi'].includes(ind)) return 1.2;
                    if (['rsi', 'bb', 'stoch'].includes(ind)) return 0.8;
                }
                return 1.0;
            };

            // ── RSI ──
            const rsiVal = 15 + sr(seed + 1) * 70;
            let rsiSig = 0,
                rsiConf = false;
            if (rsiVal < cfg.rsiOS) { rsiSig = 1;
                rsiConf = rsiVal < cfg.rsiOS - 5; } else if (rsiVal > cfg.rsiOB) { rsiSig = -1;
                rsiConf = rsiVal > cfg.rsiOB + 5; } else if (rsiVal < 45) { rsiSig = 1; } else if (rsiVal > 55) { rsiSig =
                    -1; }

            // ── MACD ──
            const macdLine = (sr(seed + 2) - 0.5) * 3;
            const macdSig_ = (sr(seed + 3) - 0.5) * 2.5;
            const macdHist = macdLine - macdSig_;
            const macdSig = macdHist > 0.04 ? 1 : macdHist < -0.04 ? -1 : 0;
            const macdConf = Math.abs(macdHist) > 0.25;

            // ── EMA ──
            const emaFast = sr(seed + 4);
            const emaSig = emaFast > 0.52 ? 1 : emaFast < 0.48 ? -1 : 0;
            const emaConf = emaFast > 0.65 || emaFast < 0.35;

            // ── VWAP ──
            const vwapPos = (sr(seed + 5) - 0.5) * 2;
            const vwapSig = vwapPos > 0.08 ? 1 : vwapPos < -0.08 ? -1 : 0;
            const vwapConf = Math.abs(vwapPos) > 0.2;

            // ── SNR ──
            const snrRand = sr(seed + 6);
            const snrStrength = sr(seed + 7);
            const snrSig = snrRand > 0.5 ? 1 : -1;
            const snrConf = snrStrength > 0.62;
            const snrTypes = ['Key Support', 'Key Resistance', 'Daily Level', '4H Zone', 'Order Block', 'Liquidity Zone'];
            const snrLabel = snrTypes[Math.floor(sr(seed + 67) * snrTypes.length)];

            // ── PA ──
            const paRand = sr(seed + 10);
            const paBias = csDir > 0 ? 0.56 : 0.44;
            const paSig = paRand > paBias ? 1 : -1;
            const paConf = Math.abs(paRand - 0.5) > 0.28;
            const paPatterns = ['Bullish Engulfing', 'Bearish Engulfing', 'Hammer', 'Shooting Star', 'Pin Bar',
                'Doji Reversal', 'Inside Bar'
            ];
            const paLabel = paPatterns[Math.floor(sr(seed + 105) * paPatterns.length)];

            // ── BOLLINGER BANDS ──
            const bbPos = sr(seed + 8);
            const bbSig = bbPos < 0.15 ? 1 : bbPos > 0.85 ? -1 : 0;
            const bbConf = bbPos < 0.1 || bbPos > 0.9;

            // ── STOCHASTIC ──
            const stochK = sr(seed + 9) * 100;
            const stochSig = stochK < 25 ? 1 : stochK > 75 ? -1 : 0;
            const stochConf = stochK < 18 || stochK > 82;

            // ── ICHIMOKU (simplified) ──
            const ichiTenkan = sr(seed + 12);
            const ichiKijun = sr(seed + 13);
            const ichiSig = ichiTenkan > ichiKijun ? 1 : -1;
            const ichiConf = Math.abs(ichiTenkan - ichiKijun) > 0.15;

            // ── FIBONACCI ──
            const fibRand = sr(seed + 14);
            const fibSig = fibRand > 0.5 ? 1 : -1;
            const fibConf = Math.abs(fibRand - 0.5) > 0.3;

            // ── MTF (higher TF trend) ──
            let mtfDir = 0,
                mtfConf = 0;
            if (settings.mtf) {
                const higherSeed = seed + 1000;
                const hTrend = sr(higherSeed + 1);
                mtfDir = hTrend > 0.55 ? 1 : hTrend < 0.45 ? -1 : 0;
                mtfConf = Math.abs(hTrend - 0.5) > 0.15 ? 1 : 0;
            }

            // ── WEIGHTED SCORE ──
            let score = (csDiff / 100) * 3.0;
            const inds = [];
            const w = cfg.w;

            // Apply dynamic weights
            const rsiW = w.rsi * weightScale('rsi');
            const macdW = w.macd * weightScale('macd');
            const emaW = w.ema * weightScale('ema');
            const vwapW = w.vwap * weightScale('vwap');
            const snrW = w.snr * weightScale('snr');
            const paW = w.pa * weightScale('pa');
            const bbW = w.bb * weightScale('bb');
            const stochW = w.stoch * weightScale('stoch');
            const ichiW = w.ichi * weightScale('ichi');
            const fibW = w.fib * weightScale('fib');

            score += rsiSig * rsiW;
            inds.push({ n: 'RSI(' + Math.round(rsiVal) + ')', s: rsiSig, conf: rsiConf });

            score += macdSig * macdW * (macdConf ? 1 : 0.7);
            inds.push({ n: 'MACD(' + macdHist.toFixed(2) + ')', s: macdSig, conf: macdConf });

            score += emaSig * emaW * (emaConf ? 1 : 0.6);
            inds.push({ n: 'EMA Trend', s: emaSig, conf: emaConf });

            score += vwapSig * vwapW * (vwapConf ? 1 : 0.7);
            inds.push({ n: 'VWAP', s: vwapSig, conf: vwapConf });

            const snrAlign = (snrSig === csDir) ? 1.3 : 0.7;
            score += snrSig * snrW * snrAlign;
            inds.push({ n: snrLabel, s: snrSig, conf: snrConf });

            score += paSig * paW * (paConf ? 1 : 0.6);
            inds.push({ n: paLabel, s: paSig, conf: paConf });

            score += bbSig * bbW * (bbConf ? 1 : 0.6);
            inds.push({ n: 'Bollinger', s: bbSig, conf: bbConf });

            score += stochSig * stochW * (stochConf ? 1 : 0.6);
            inds.push({ n: 'Stoch(' + Math.round(stochK) + ')', s: stochSig, conf: stochConf });

            score += ichiSig * ichiW * (ichiConf ? 1 : 0.6);
            inds.push({ n: 'Ichimoku', s: ichiSig, conf: ichiConf });

            score += fibSig * fibW * (fibConf ? 1 : 0.6);
            inds.push({ n: 'Fibonacci', s: fibSig, conf: fibConf });

            // MTF adjustment
            if (settings.mtf && mtfDir !== 0) {
                const mtfFactor = mtfConf * 1.5;
                score += mtfDir * mtfFactor;
                inds.push({ n: 'MTF Trend', s: mtfDir, conf: mtfConf > 0.5 });
            }

            // ── FINAL DIRECTION ──
            let rawDir = score > 0 ? 1 : -1;
            if (csSpread > 30 && rawDir !== csDir) rawDir = csDir;
            const isUp = rawDir > 0;
            const dir = isUp ? 1 : -1;

            const confirming = inds.filter(i => i.s === dir);
            const confRate = confirming.length / inds.length;
            let qual = Math.min(99, Math.round(55 + (confRate * 44)));
            // Boost for MTF alignment
            if (settings.mtf && mtfDir === dir) qual = Math.min(99, qual + 5);
            // Penalize if high volatility and low trend confirmation
            if (isTrending && confRate < 0.5) qual = Math.max(50, qual - 10);

            // ── BLOCK GATE ──
            const minConf = { 1: 5, 2: 5, 3: 6, 5: 6 }; // increased for more accuracy
            const needed = minConf[selectedTF] || 5;
            const blocked = confirming.length < needed || qual < 70;

            // ── VOLATILITY ──
            const volLabel = vol > 0.65 ? 'HIGH' : vol > 0.4 ? 'MEDIUM' : 'LOW';

            return {
                blocked: blocked,
                isUp: isUp,
                qual: qual,
                rsi: Math.round(rsiVal),
                csSpread: Math.round(csSpread),
                csDir: csDir,
                c1: c1,
                c2: c2,
                v1: Math.round(v1),
                v2: Math.round(v2),
                confirming: confirming,
                total: inds.length,
                snrLabel: snrLabel,
                paLabel: paLabel,
                vol: volLabel,
                trend: isUp ? 'BULLISH' : 'BEARISH',
                mtfDir: mtfDir
            };
        }

        // ─── BEST PAIR TRACKING ───
        let pairStats = {};

        function updateBestPair() {
            const p = currentSymbol;
            if (!pairStats[p]) pairStats[p] = { total: 0, wins: 0 };
            pairStats[p].total++;
            // Simulate win based on confidence
            const lastQual = parseInt(confidenceVal.textContent) || 70;
            if (lastQual > 70 && Math.random() < 0.85) pairStats[p].wins++;
            let bestPair = '';
            let bestRate = 0;
            for (const [pair, stat] of Object.entries(pairStats)) {
                if (stat.total >= 3) {
                    const rate = stat.wins / stat.total;
                    if (rate > bestRate) { bestRate = rate;
                        bestPair = pair; }
                }
            }
            bestPairLabel.textContent = bestPair ? 'Best: ' + bestPair + ' (' + Math.round(bestRate * 100) + '%)' :
            'Best: --';
            totalSignalsLabel.textContent = 'Signals: ' + totalSignals;
        }

        // ─── EXPIRY ───
        function startExpiry(totalSec, isUp) {
            clearInterval(expiryTimer);
            signalActive = true;
            let rem = totalSec;

            function update() {
                const m = Math.floor(rem / 60);
                const s = rem % 60;
                const str = m + ':' + (s < 10 ? '0' + s : s);
                timerLeft.textContent = str;
                const color = rem <= 10 ? 'var(--red)' : rem <= 30 ? 'var(--gold)' : (isUp ? 'var(--green)' : 'var(--red)');
                timerLeft.style.color = color;
                if (rem <= 0) {
                    clearInterval(expiryTimer);
                    expiryTimer = null;
                    signalActive = false;
                    btnGenerate.disabled = false;
                    btnGenerate.innerText = '⚡ GENERATE SIGNAL';
                    signalStatus.textContent = '● EXPIRED';
                    // Auto re-entry if enabled
                    if (settings.reentry && !isGenerating) {
                        setTimeout(() => {
                            if (!signalActive && !isGenerating) {
                                startGenerationProcess();
                            }
                        }, 3000);
                    } else {
                        setTimeout(() => {
                            if (!signalActive) {
                                resetSignal();
                                signalStatus.textContent = '● IDLE';
                            }
                        }, 2000);
                    }
                }
                rem--;
            }
            update();
            expiryTimer = setInterval(update, 1000);
        }

        function clearAllTimers() {
            clearInterval(expiryTimer);
            expiryTimer = null;
            signalActive = false;
        }

        (function() {
            const canvas = document.getElementById('bg-canvas');
            const ctx = canvas.getContext('2d');
            let w, h, particles = [];

            function resize() {
                w = canvas.width = window.innerWidth;
                h = canvas.height = window.innerHeight;
            }
            window.addEventListener('resize', resize);
            resize();

            class Particle {
                constructor() { this.reset(); }
                reset() {
                    this.x = Math.random() * w;
                    this.y = h + Math.random() * 100;
                    this.speed = Math.random() * 1.8 + 0.8;
                    this.size = Math.random() * 1.8 + 0.8;
                    this.type = Math.random() > 0.8 ? 'candle' : 'star';
                    this.color = Math.random() > 0.5 ? '#00ff66' : '#ff2a55';
                    this.opacity = Math.random() * 0.4 + 0.08;
                    this.height = Math.random() * 18 + 8;
                }
                update() { this.y -= this.speed; if (this.y < -50) this.reset(); }
                draw() {
                    ctx.globalAlpha = this.opacity;
                    if (this.type === 'candle') {
                        ctx.fillStyle = this.color;
                        ctx.fillRect(this.x, this.y, this.size * 2, this.height);
                        ctx.fillRect(this.x + this.size - 0.5, this.y - 4, 1, this.height + 8);
                    } else {
                        ctx.fillStyle = '#ffffff';
                        ctx.beginPath();
                        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                        ctx.fill();
                    }
                    ctx.globalAlpha = 1;
                }
            }
            for (let i = 0; i < 50; i++) particles.push(new Particle());

            function animate() {
                ctx.clearRect(0, 0, w, h);
                particles.forEach(p => { p.update();
                    p.draw(); });
                requestAnimationFrame(animate);
            }
            animate();
